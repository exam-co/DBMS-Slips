# CREATE TABLES

```sql id="x8m3tp"
-- =========================
-- TRAIN TABLE
-- =========================

CREATE TABLE Train (
    Train_no INT PRIMARY KEY,
    train_name VARCHAR(20),
    depart_time TIME,
    arrival_time TIME,
    source_stn VARCHAR(20),
    dest_stn VARCHAR(20),
    no_of_res_bogies INT,
    bogie_capacity INT
);

-- =========================
-- PASSENGER TABLE
-- =========================

CREATE TABLE Passenger (
    Passenger_id INT PRIMARY KEY,
    passenger_name VARCHAR(20),
    address VARCHAR(30),
    age INT,
    gender CHAR(1)
);

-- =========================
-- TICKET TABLE
-- =========================

CREATE TABLE Ticket (
    ticket_no INT PRIMARY KEY,
    Train_no INT,
    Passenger_id INT,
    bogie_no INT,
    no_of_berths INT,
    tdate DATE,
    ticket_amt DECIMAL(7,2),
    status CHAR(1),

    FOREIGN KEY (Train_no)
    REFERENCES Train(Train_no),

    FOREIGN KEY (Passenger_id)
    REFERENCES Passenger(Passenger_id)
);
```

# 1) List train names with departure and arrival times

```sql id="r6m1xn"
SELECT train_name,
       depart_time,
       arrival_time
FROM Train;
```

---

# 2) Find names and addresses of passengers older than 60

```sql id="t4q8vp"
SELECT passenger_name,
       address
FROM Passenger
WHERE age > 60;
```

---

# 3) Display source and destination stations of trains having more than 10 reserved bogies

```sql id="m9w2tr"
SELECT source_stn,
       dest_stn
FROM Train
WHERE no_of_res_bogies > 10;
```

---

# 4) Display names of all male passengers

```sql id="k3x7qn"
SELECT passenger_name
FROM Passenger
WHERE gender = 'M';
```

# Q2 SOLUTION

# 1) Trigger to restrict bogie capacity to 30

```sql id="p7v4xm"
-- =========================
-- TRIGGER FUNCTION
-- =========================

CREATE OR REPLACE FUNCTION check_bogie_capacity()
RETURNS TRIGGER AS
$$
BEGIN

    IF NEW.bogie_capacity > 30 THEN
        RAISE EXCEPTION 'Bogie capacity cannot be greater than 30';
    END IF;

    RETURN NEW;

END;
$$
LANGUAGE plpgsql;
```

---

# CREATE TRIGGER

```sql id="u2r9wt"
CREATE TRIGGER trg_check_bogie_capacity
BEFORE INSERT OR UPDATE
ON Train
FOR EACH ROW
EXECUTE FUNCTION check_bogie_capacity();
```

# 2) Stored Function to calculate total ticket amount on 12/12/2019

```sql id="z1v6rk"
CREATE OR REPLACE FUNCTION total_ticket_amount()
RETURNS DECIMAL AS
$$
DECLARE

    total DECIMAL;

BEGIN

    SELECT SUM(ticket_amt)
    INTO total
    FROM Ticket
    WHERE tdate = '2019-12-12';

    RETURN total;

END;
$$
LANGUAGE plpgsql;
```

---

# EXECUTE FUNCTION

