# CREATE TABLES

```sql id="q8m4tp"
-- =========================
-- STUDENT TABLE
-- =========================

CREATE TABLE Student (
    student_id INT PRIMARY KEY,
    name VARCHAR(30),
    age INT
);

-- =========================
-- COURSE TABLE
-- =========================

CREATE TABLE Course (
    course_id INT PRIMARY KEY,
    course_name VARCHAR(30),
    credits INT
);

-- =========================
-- ENROLLMENT TABLE
-- =========================

CREATE TABLE Enrollment (
    student_id INT,
    course_id INT,
    enrollment_date DATE,

    PRIMARY KEY (student_id, course_id),

    FOREIGN KEY (student_id)
    REFERENCES Student(student_id),

    FOREIGN KEY (course_id)
    REFERENCES Course(course_id)
);
```

```sql id="p6w2qx"
SELECT e.student_id
FROM Enrollment e
JOIN Course c
ON e.course_id = c.course_id
WHERE c.course_name = 'B.C.A';
```

# Q2 SOLUTION

# 1) Function to display details of student whose name starts with 'P'

```sql id="r9m3ty"
CREATE OR REPLACE FUNCTION student_name_p()
RETURNS TABLE (
    student_id INT,
    student_name VARCHAR,
    age INT
)
AS
$$
BEGIN

    RETURN QUERY

    SELECT
        s.student_id,
        s.name,
        s.age
    FROM Student s
    WHERE s.name LIKE 'P%';

END;
$$
LANGUAGE plpgsql;
```

# 2) Create View to show each student's name and enrolled courses

```sql id="f5q8wr"
CREATE VIEW student_course_view AS
SELECT s.name,
       c.course_name
FROM Student s
JOIN Enrollment e
ON s.student_id = e.student_id
JOIN Course c
ON e.course_id = c.course_id;
```

---

# DISPLAY VIEW

```sql id="n4v1tp"
SELECT * FROM student_course_view;
```
