# CREATE TABLES

```sql id="z4k8np"
-- =========================
-- DOCTOR TABLE
-- =========================

CREATE TABLE Doctor (
    did INT PRIMARY KEY,
    dname VARCHAR(30),
    specialization VARCHAR(30)
);

-- =========================
-- PATIENT TABLE
-- =========================

CREATE TABLE Patient (
    pid INT PRIMARY KEY,
    pname VARCHAR(30),
    age INT,
    did INT,

    FOREIGN KEY (did)
    REFERENCES Doctor(did)
);
```

# Q1 SOLUTION

# 1) List patient names of doctor 'Dr. Sharma'

```sql id="i2n7vy"
SELECT p.pname
FROM Patient p
JOIN Doctor d
ON p.did = d.did
WHERE d.dname = 'Dr. Sharma';
```

---

# 2) List doctors all details

```sql id="w9e4qs"
SELECT *
FROM Doctor;
```

---

# 3) Update specialization of 'Dr. Gupta' to 'Cardiologist'

```sql id="m5r2ux"
UPDATE Doctor
SET specialization = 'Cardiologist'
WHERE dname = 'Dr. Gupta';
```

---

# 4) Delete all patients of 'Dr. Agarwal'

```sql id="j3x8cp"
DELETE FROM Patient
WHERE did = (
    SELECT did
    FROM Doctor
    WHERE dname = 'Dr. Agarwal'
);
```

# 1) Create Views

## (a) Display doctor details who check patient 'Mr.P.V.Joshi'

```sql id="t6v0zn"
CREATE VIEW doctor_patient_view AS
SELECT d.*
FROM Doctor d
JOIN Patient p
ON d.did = p.did
WHERE p.pname = 'Mr. P.V.Joshi';
```

# 2) Cursor to display doctor details along with patient

```sql id="v4n7yt"
CREATE OR REPLACE FUNCTION doctor_patient_cursor()
RETURNS VOID AS
$$
DECLARE

    rec RECORD;

    cur CURSOR FOR
    SELECT d.dname,
           d.specialization,
           p.pname
    FROM Doctor d
    JOIN Patient p
    ON d.did = p.did;

BEGIN

    OPEN cur;

    LOOP

        FETCH cur INTO rec;

        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'Doctor: %, Specialization: %, Patient: %',
        rec.dname,
        rec.specialization,
        rec.pname;

    END LOOP;

    CLOSE cur;

END;
$$
LANGUAGE plpgsql;
```
