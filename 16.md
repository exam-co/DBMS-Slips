# CREATE TABLES

```sql id="h7x2qp"
-- =========================
-- BRANCH TABLE
-- =========================

CREATE TABLE Branch (
    B_id INT PRIMARY KEY,
    Bname VARCHAR(30),
    Bcity VARCHAR(30)
);

-- =========================
-- CUSTOMER TABLE
-- =========================

CREATE TABLE Customer (
    C_no INT PRIMARY KEY,
    Cname VARCHAR(30),
    Caddress VARCHAR(50),
    City VARCHAR(30)
);

-- =========================
-- LOAN_APPLICATION TABLE
-- =========================

CREATE TABLE Loan_Application (
    L_no INT PRIMARY KEY,
    L_amt_required DECIMAL(10,2),
    L_amt_approved DECIMAL(10,2),
    L_date DATE
);

-- =========================
-- TERNARY RELATION TABLE
-- =========================

CREATE TABLE Branch_Customer_Loan (
    B_id INT,
    C_no INT,
    L_no INT,

    PRIMARY KEY (B_id, C_no, L_no),

    FOREIGN KEY (B_id)
    REFERENCES Branch(B_id),

    FOREIGN KEY (C_no)
    REFERENCES Customer(C_no),

    FOREIGN KEY (L_no)
    REFERENCES Loan_Application(L_no)
);
```

# Q1 SOLUTION

# 1) List names of customers for the 'Aundh' branch

```sql id="7qf9uw"
SELECT c.Cname
FROM Customer c
JOIN Branch_Customer_Loan bcl
ON c.C_no = bcl.C_no
JOIN Branch b
ON bcl.B_id = b.B_id
WHERE b.Bname = 'Aundh';
```

---

# 2) Find maximum loan amount approved

```sql id="2n5rxe"
SELECT MAX(L_amt_approved) AS maximum_approved_loan
FROM Loan_Application;
```

---

# 3) Count number of loan applications received by 'M.G. Road' branch

```sql id="6m0jpt"
SELECT COUNT(*) AS total_applications
FROM Branch_Customer_Loan bcl
JOIN Branch b
ON bcl.B_id = b.B_id
```

# 2) Stored Function using Cursor

```sql id="c5z1wo"
CREATE OR REPLACE FUNCTION customer_branch_by_date(input_date DATE)
RETURNS VOID AS
$$
DECLARE

    rec RECORD;

    cur CURSOR FOR
    SELECT c.Cname,
           b.Bname
    FROM Customer c
    JOIN Branch_Customer_Loan bcl
    ON c.C_no = bcl.C_no
    JOIN Branch b
    ON bcl.B_id = b.B_id
    JOIN Loan_Application l
    ON bcl.L_no = l.L_no
    WHERE l.L_date = input_date;

BEGIN

    OPEN cur;

    LOOP

        FETCH cur INTO rec;

        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'Customer: %, Branch: %',
        rec.Cname,
        rec.Bname;

    END LOOP;

    CLOSE cur;

END;
$$
LANGUAGE plpgsql;
```
