# CREATE TABLES

```sql id="g7m2xp"
-- =========================
-- STUDENT TABLE
-- =========================

CREATE TABLE Student (
    student_id INT PRIMARY KEY,
    name VARCHAR(30),
    age INT
);

-- =========================
-- COURSE TABLE
-- =========================

CREATE TABLE Course (
    course_id INT PRIMARY KEY,
    course_name VARCHAR(40),
    credits INT
);

-- =========================
-- ENROLLMENT TABLE
-- =========================

CREATE TABLE Enrollment (
    student_id INT,
    course_id INT,
    enrollment_date DATE,

    PRIMARY KEY (student_id, course_id),

    FOREIGN KEY (student_id)
    REFERENCES Student(student_id),

    FOREIGN KEY (course_id)
    REFERENCES Course(course_id)
);
```

# Q1 SOLUTION

# 1) List names of all students enrolled in B.Sc(Comp.Sci) course

```sql id="p6x1wn"
SELECT s.name
FROM Student s
JOIN Enrollment e
ON s.student_id = e.student_id
JOIN Course c
ON e.course_id = c.course_id
WHERE c.course_name = 'B.Sc(Comp.Sci)';
```

---

# 2) Show all courses with more than 3 credits

```sql id="m8t4qy"
SELECT *
FROM Course
WHERE credits > 3;
```

# Q2 SOLUTION

# 1) Function to display count of student

```sql id="u1q9mx"
CREATE OR REPLACE FUNCTION student_count()
RETURNS INT AS
$$
DECLARE

    total INT;

BEGIN

    SELECT COUNT(*)
    INTO total
    FROM Student;

    RETURN total;

END;
$$
LANGUAGE plpgsql;
```

---

# EXECUTE FUNCTION

```sql id="x7m4tr"

```

# 2) Create a View to show each student's name and enrolled courses

```sql id="r8v5qn"
CREATE VIEW student_course_view AS
SELECT s.name,
       c.course_name
FROM Student s
JOIN Enrollment e
ON s.student_id = e.student_id
JOIN Course c
ON e.course_id = c.course_id;
```

# DISPLAY VIEW
