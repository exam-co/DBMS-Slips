# CREATE TABLES

```sql id="x4m8tp"
-- =========================
-- BRANCH TABLE
-- =========================

CREATE TABLE Branch (
    bid INT PRIMARY KEY,
    brname VARCHAR(30),
    brcity VARCHAR(20)
);

-- =========================
-- CUSTOMER TABLE
-- =========================

CREATE TABLE Customer (
    cno INT PRIMARY KEY,
    cname VARCHAR(20),
    caddr VARCHAR(35),
    city VARCHAR(20)
);

-- =========================
-- LOAN_APPLICATION TABLE
-- =========================

CREATE TABLE Loan_Application (
    lno INT PRIMARY KEY,
    lamtrequired MONEY,
    lamtapproved MONEY,
    l_date DATE,
    bid INT,
    cno INT,

    FOREIGN KEY (bid)
    REFERENCES Branch(bid),

    FOREIGN KEY (cno)
    REFERENCES Customer(cno)
);
```

# Q1 SOLUTION

# 1) List names of customers who received loan less than requirement

```sql id="f8t3xn"
SELECT c.cname
FROM Customer c
JOIN Loan_Application l
ON c.cno = l.cno
WHERE l.lamtapproved < l.lamtrequired;
```

---

# 2) Find maximum loan amount approved

```sql id="v5m1qp"
SELECT MAX(lamtapproved) AS maximum_loan_approved
FROM Loan_Application;
```

---

# 3) Find total loan amount sanctioned by 'Deccan' branch

```sql id="m2x7wr"
SELECT SUM(l.lamtapproved) AS total_loan_amount
FROM Loan_Application l
JOIN Branch b
ON l.bid = b.bid
WHERE b.brname = 'Deccan';
```

---

# 4) Count loan applications received by 'M.G.ROAD' branch

```sql id="k9v4ty"
SELECT COUNT(*) AS total_applications
FROM Loan_Application l
JOIN Branch b
ON l.bid = b.bid
WHERE b.brname = 'M.G.ROAD';
```

# Q2 SOLUTION

# 1) Create View to display customers who received loan less than requirement

```sql id="u3q8zp"
CREATE VIEW customer_loan_view AS
SELECT c.*,
       l.lamtrequired,
       l.lamtapproved
FROM Customer c
JOIN Loan_Application l
ON c.cno = l.cno
WHERE l.lamtapproved < l.lamtrequired;
```

---

# DISPLAY VIEW


# 2) Stored Function to display customer details who applied in more than one branch

```sql id="p4t9vn"
CREATE OR REPLACE FUNCTION multiple_branch_customers()
RETURNS TABLE (
    customer_name VARCHAR,
    city_name VARCHAR,
    total_branches BIGINT
)
AS
$$
BEGIN

    RETURN QUERY

    SELECT
        c.cname,
        c.city,
        COUNT(DISTINCT l.bid)

    FROM Customer c
    JOIN Loan_Application l
    ON c.cno = l.cno

    GROUP BY c.cname, c.city

    HAVING COUNT(DISTINCT l.bid) > 1;

END;
$$
LANGUAGE plpgsql;
```
