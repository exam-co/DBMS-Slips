# CREATE TABLES

```sql id="m5x8tp"
-- =========================
-- ROUTE TABLE
-- =========================

CREATE TABLE Route (
    route_no INT PRIMARY KEY,
    source VARCHAR(20),
    destination VARCHAR(20),
    no_of_stations INT
);

-- =========================
-- DRIVER TABLE
-- =========================

CREATE TABLE Driver (
    driver_no INT PRIMARY KEY,
    driver_name VARCHAR(20),
    license_no INT,
    address VARCHAR(20),
    d_age INT,
    salary FLOAT
);

-- =========================
-- BUS TABLE
-- =========================

CREATE TABLE Bus (
    bus_no INT PRIMARY KEY,
    b_capacity INT,
    depot_name VARCHAR(20),
    route_no INT,
    driver_no INT,

    FOREIGN KEY (route_no)
    REFERENCES Route(route_no),

    FOREIGN KEY (driver_no)
    REFERENCES Driver(driver_no)
);
```

# Q1 SOLUTION

# 1) Display buses with capacity greater than 50

```sql id="x4m9tp"
SELECT *
FROM Bus
WHERE b_capacity > 50;
```

---

# 2) List all routes starting from 'Delhi'

```sql id="r8v3wn"
SELECT *
FROM Route
WHERE source = 'Delhi';
```

---

# 3) Display names and license numbers of drivers earning more than 25000

```sql id="k2q7vx"
SELECT driver_name,
       license_no
FROM Driver
WHERE salary > 25000;
```

---

# 4) List all drivers of Kothrud bus depot

```sql id="p6w1tm"
SELECT d.driver_name
FROM Driver d
JOIN Bus b
ON d.driver_no = b.driver_no
WHERE b.depot_name = 'Kothrud';
```

# Q2 SOLUTION

# 1) Stored Function to accept route number and display bus information

```sql id="u9x4rp"
CREATE OR REPLACE FUNCTION bus_info_route(r_no INT)
RETURNS TABLE (
    bus_number INT,
    capacity INT,
    depot VARCHAR
)
AS
$$
BEGIN

    RETURN QUERY

    SELECT
        b.bus_no,
        b.b_capacity,
        b.depot_name
    FROM Bus b
    WHERE b.route_no = r_no;

END;
$$
LANGUAGE plpgsql;
```

# 2) Trigger to prevent deleting drivers living in 'Satara'

```sql id="t7w2xn"
-- =========================
-- TRIGGER FUNCTION
-- =========================

CREATE OR REPLACE FUNCTION prevent_delete_satara_driver()
RETURNS TRIGGER AS
$$
BEGIN

    IF OLD.address = 'Satara' THEN
        RAISE EXCEPTION 'Drivers living in Satara cannot be deleted';
    END IF;

    RETURN OLD;

END;
$$
LANGUAGE plpgsql;
```

---

# CREATE TRIGGER

```sql id="v5q9tr"
CREATE TRIGGER trg_prevent_delete_driver
BEFORE DELETE
ON Driver
FOR EACH ROW
EXECUTE FUNCTION prevent_delete_satara_driver();
```
