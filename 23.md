# CREATE TABLES

```sql id="m4x8tp"
-- =========================
-- BUS TABLE
-- =========================

CREATE TABLE Bus (
    bus_no INT PRIMARY KEY,
    capacity INT,
    depot_name VARCHAR(20)
);

-- =========================
-- ROUTE TABLE
-- =========================

CREATE TABLE Route (
    route_no INT PRIMARY KEY,
    source VARCHAR(20),
    destination VARCHAR(20),
    no_of_stations INT
);

-- =========================
-- DRIVER TABLE
-- =========================

CREATE TABLE Driver (
    driver_no INT PRIMARY KEY,
    driver_name VARCHAR(20),
    license_no VARCHAR(20),
    address VARCHAR(20),
    d_age INT,
    salary FLOAT
);

-- =========================
-- BUS_ROUTE TABLE
-- =========================

CREATE TABLE Bus_Route (
    bus_no INT,
    route_no INT,

    PRIMARY KEY (bus_no, route_no),

    FOREIGN KEY (bus_no)
    REFERENCES Bus(bus_no),

    FOREIGN KEY (route_no)
    REFERENCES Route(route_no)
);

-- =========================
-- BUS_DRIVER TABLE
-- =========================

CREATE TABLE Bus_Driver (
    bus_no INT,
    driver_no INT,

    PRIMARY KEY (bus_no, driver_no),

    FOREIGN KEY (bus_no)
    REFERENCES Bus(bus_no),

    FOREIGN KEY (driver_no)
    REFERENCES Driver(driver_no)
);
```

# Q2 SOLUTION

# 1) Stored Function to accept depot name and display driver details having age > 50

```sql id="f9w2rm"
CREATE OR REPLACE FUNCTION driver_details(dep_name VARCHAR)
RETURNS TABLE (
    driver_name VARCHAR,
    age INT,
    depot VARCHAR
)
AS
$$
BEGIN

    RETURN QUERY

    SELECT
        d.driver_name,
        d.d_age,
        b.depot_name
    FROM Driver d
    JOIN Bus_Driver bd
    ON d.driver_no = bd.driver_no
    JOIN Bus b
    ON bd.bus_no = b.bus_no
    WHERE b.depot_name = dep_name
    AND d.d_age > 50;

END;
$$
LANGUAGE plpgsql;
```

# 2) Trigger to prevent deleting drivers living in 'Pune'

```sql id="w2m6qy"
-- =========================
-- TRIGGER FUNCTION
-- =========================

CREATE OR REPLACE FUNCTION prevent_delete_driver()
RETURNS TRIGGER AS
$$
BEGIN

    IF OLD.address = 'Pune' THEN
        RAISE EXCEPTION 'Drivers living in Pune cannot be deleted';
    END IF;

    RETURN OLD;

END;
$$
LANGUAGE plpgsql;
```

---

# CREATE TRIGGER

```sql id="t7v3xn"
CREATE TRIGGER trg_prevent_delete
BEFORE DELETE
ON Driver
FOR EACH ROW
EXECUTE FUNCTION prevent_delete_driver();
```
