# CREATE TABLES

```sql id="p8m4wx"
-- =========================
-- PRODUCT TABLE
-- =========================

CREATE TABLE Product (
    pid INT PRIMARY KEY,
    pname VARCHAR(30),
    price DECIMAL(10,2)
);

-- =========================
-- ORDER TABLE
-- =========================

CREATE TABLE Orders (
    oid INT PRIMARY KEY,
    orderdate DATE,
    customer VARCHAR(30)
);

-- =========================
-- ORDER_PRODUCT TABLE
-- =========================

CREATE TABLE Order_Product (
    oid INT,
    pid INT,

    PRIMARY KEY (oid, pid),

    FOREIGN KEY (oid)
    REFERENCES Orders(oid),

    FOREIGN KEY (pid)
    REFERENCES Product(pid)
);
```

FROM Product p
JOIN Order_Product op
ON p.pid = op.pid
WHERE op.oid = 101;

## (b) Display order with highest demand

```sql id="c2m7tv"
CREATE VIEW highest_demand_order AS
SELECT oid,
       COUNT(pid) AS total_products
FROM Order_Product
GROUP BY oid
ORDER BY total_products DESC;
```

---

## DISPLAY VIEW

```sql id="v3r9yx"

```

# 2) Function to display all product details with customer

```sql id="h7p4zn"
CREATE OR REPLACE FUNCTION product_customer_details()
RETURNS TABLE (
    customer_name VARCHAR,
    product_name VARCHAR,
    product_price DECIMAL
)
AS
$$
BEGIN

    RETURN QUERY

    SELECT
        o.customer,
        p.pname,
        p.price
    FROM Orders o
    JOIN Order_Product op
    ON o.oid = op.oid
    JOIN Product p
    ON op.pid = p.pid;

END;
$$
LANGUAGE plpgsql;
```
